function chatbot(input) { const text = input.toLowerCase(); if (text.includes("مرحبا") || text.includes("أهلا")) { return "مرحبًا! كيف يمكنني مساعدتك اليوم؟"; } if (text.includes("كيف حالك")) { return "أنا بخير، شكرًا لسؤالك! كيف يمكنني مساعدتك؟"; } if (text.includes("ساعدني") || text.includes("مساعدة")) { return "بالطبع! أخبرني ماذا تريد أن تعرف أو تحتاج."; } if (text.includes("من أنت")) { return "أنا شات بوت بسيط، هنا لأساعدك في أي سؤال."; } if (text.includes("شكراً") || text.includes("شكرًا")) { return "على الرحب والسعة!"; } // لو ما عرف يرد، يعطي رد عام return "لم أفهم سؤالك جيدًا، هل يمكنك توضيح أكثر؟"; } // مثال تجربة: console.log(chatbot("مرحبا")); // مرحبًا! كيف يمكنني مساعدتك اليوم؟ console.log(chatbot("كيف حالك؟")); // أنا بخير، شكرًا لسؤالك! كيف يمكنني مساعدتك؟ console.log(chatbot("ساعدني")); // بالطبع! أخبرني ماذا تريد أن تعرف أو تحتاج. console.log(chatbot("من أنت؟")); // أنا شات بوت بسيط، هنا لأساعدك في أي سؤال. console.log(chatbot("شكراً جزيلاً")); // على الرحب والسعة! console.log(chatbot("ماذا تفعل؟")); // لم أفهم سؤالك جيدًا، هل يمكنك توضيح أكثر؟
